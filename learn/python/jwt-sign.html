Sign a JWT

import jwt
token = jwt.encode({'sub': 'user-1'}, secret, algorithm='HS256')
try:
    decoded = jwt.decode(token, secret, algorithms=['HS256'])
except jwt.InvalidTokenError as e:
    print('jwt sign token python:', e)
